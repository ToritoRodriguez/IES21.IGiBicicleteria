-- Creamos la tabla personas
CREATE TABLE personas (
    id INT AUTO_INCREMENT,
    nombre VARCHAR(45) NOT NULL,
    apellido VARCHAR(45) NOT NULL,
    dni VARCHAR(20) NOT NULL,
    telefono VARCHAR(30),
    email VARCHAR(55),
    PRIMARY KEY (id)
);

-- Creamos la tabla clientes
CREATE TABLE clientes (
    id INT AUTO_INCREMENT,
    id_persona INT NOT NULL,
    codigo VARCHAR(25) NOT NULL,
    cuil VARCHAR(14),
    PRIMARY KEY (id),
    FOREIGN KEY (id_persona) REFERENCES personas(id)
);

-- Modificamos la tabla vendedores
CREATE TABLE vendedores (
    id INT AUTO_INCREMENT,
    id_persona INT NOT NULL,
    codigo VARCHAR(25) NOT NULL,
    sucursal VARCHAR(20),
    cuit VARCHAR(14), 
    PRIMARY KEY (id),
    FOREIGN KEY (id_persona) REFERENCES personas(id)
);

-- Creamos la tabla proveedores
CREATE TABLE proveedores (
    id INT AUTO_INCREMENT,
    id_persona INT NOT NULL,
    codigo VARCHAR(25) NOT NULL,
    nombre_fantasia VARCHAR(45),
    cuit VARCHAR(14) NOT NULL,
    PRIMARY KEY (id),
    FOREIGN KEY (id_persona) REFERENCES personas(id)
);

-- Creamos la tabla categorias
CREATE TABLE categorias (
    id INT AUTO_INCREMENT,
    codigo VARCHAR(25) NOT NULL,
    categoria VARCHAR(20) NOT NULL,
    tipo VARCHAR(10) NOT NULL,
    PRIMARY KEY (id)
);

-- Creamos la tabla marcas
CREATE TABLE marcas (
    id INT AUTO_INCREMENT,
    codigo VARCHAR(25) NOT NULL,
    marca VARCHAR(30) NOT NULL,
    PRIMARY KEY (id)
);

-- Creamos la tabla modelos
CREATE TABLE modelos (
    id INT AUTO_INCREMENT,
    id_marca INT NOT NULL,
    codigo VARCHAR(25) NOT NULL,
    modelo VARCHAR(15) NOT NULL,
    descripcion TEXT NOT NULL,
    rodado VARCHAR(15),
    PRIMARY KEY (id),
    FOREIGN KEY (id_marca) REFERENCES marcas(id)
);

-- Creamos la tabla productos
CREATE TABLE productos (
    id INT AUTO_INCREMENT,
    id_categoria INT NOT NULL,
    id_modelo INT NOT NULL,
    id_proveedor INT NOT NULL,
    codigo VARCHAR(25) NOT NULL,
    cantidad INT NOT NULL,
    nombre TEXT NOT NULL,
    descripcion TEXT NOT NULL,
    precio NUMERIC(18,2),
    path_imagen TEXT,
    PRIMARY KEY (id),
    FOREIGN KEY (id_categoria) REFERENCES categorias(id),
    FOREIGN KEY (id_modelo) REFERENCES modelos(id),
    FOREIGN KEY (id_proveedor) REFERENCES proveedores(id)
);

-- Creamos la tabla pedido
CREATE TABLE pedido (
    id INT AUTO_INCREMENT PRIMARY KEY,
    fecha DATE NOT NULL,
    id_vendedor INT NOT NULL,
    id_cliente INT NOT NULL,
    codigo VARCHAR(25) NOT NULL,
    estado VARCHAR(50) NOT NULL,
    total_pedido DECIMAL(10, 2) NOT NULL,
    FOREIGN KEY (id_vendedor) REFERENCES vendedores(id), 
    FOREIGN KEY (id_cliente) REFERENCES clientes(id)      
);

-- Creamos la tabla Detalle_Pedido
CREATE TABLE detalle_pedido (
    id INT AUTO_INCREMENT PRIMARY KEY,
    id_pedido INT NOT NULL,
    id_producto INT NOT NULL,
    codigo VARCHAR(25) NOT NULL,
    cantidad INT NOT NULL,
    precio DECIMAL(10, 2) NOT NULL,   
    descuento INT DEFAULT NULL,      
    total DECIMAL(10, 2) NOT NULL,    
    FOREIGN KEY (id_pedido) REFERENCES pedido(id),
    FOREIGN KEY (id_producto) REFERENCES productos(id)
);
